```typescript
/** Extract the storage path from a result_url.
 *  Handles three formats:
 *  1. Bare storage path (e.g. "jobs/{jobId}/view/variant.png") — return as-is
 *  2. Supabase signed/public URL containing "generated-images/" — extract object path after bucket name
 *  3. Other https:// URLs — return null (external, not our storage)
 */
export function extractPathFromUrl(url: string): string | null {
  if (!url) return null;
  // Bare storage path — not a URL, return directly
  if (!url.startsWith("https://")) return url || null;
  // Supabase URL — extract path after bucket name
  const marker = "generated-images/";
  const idx = url.indexOf(marker);
  if (idx === -1) return null;
  let path = url.substring(idx + marker.length);
  const qIdx = path.indexOf("?");
  if (qIdx !== -1) path = path.substring(0, qIdx);
  return path || null;
}
```

This ensures:
- `"jobs/abc/front/original.png"` returns `"jobs/abc/front/original.png"` (bare path, new format from plan 02-02)
- `"https://...generated-images/user/run/img.png?token=..."` returns `"user/run/img.png"` (legacy signed URL)
- `"https://external-cdn.com/img.png"` returns `null` (external URL, not our storage)
